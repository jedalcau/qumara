DELIMITER $$

CREATE TRIGGER eliminar
    AFTER DELETE
    ON paciente FOR EACH ROW
BEGIN
    INSERT INTO paciente_delete VALUES (null,old.dni,old.nombre,old.apellidos,old.email,old.fecnac,old.sexo,old.direccion,old.distrito,old.provincia,old.ciudad,old.codpostal,old.telefono,old.avatar,old.status,old.fecCreate,now());
END$$    

DELIMITER ;


CREATE TABLE `paciente_delete` (
  `idpac` int(11) NOT NULL AUTO_INCREMENT,
  `dni` varchar(11) NOT NULL,
  `nombre` varchar(80) NOT NULL,
  `apellidos` varchar(150) NOT NULL,
  `email` varchar(150) NOT NULL,
  `fecnac` varchar(10) NOT NULL,
  `sexo` varchar(12) NOT NULL,
  `direccion` varchar(200) DEFAULT NULL,
  `distrito` varchar(40) DEFAULT NULL,
  `provincia` varchar(30) DEFAULT NULL,
  `ciudad` varchar(30) DEFAULT NULL,
  `codpostal` varchar(20) DEFAULT NULL,
  `telefono` varchar(40) DEFAULT NULL,
  `avatar` varchar(200) DEFAULT NULL,
  `status` tinyint(1) DEFAULT NULL,
  `fecCreate` datetime DEFAULT NULL,
  `fecDelete` datetime DEFAULT NULL,
  PRIMARY KEY (`idpac`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
